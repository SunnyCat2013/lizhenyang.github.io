# 91. Decode Ways 

```
def numDecodings(self, s):
    """
    :type s: str
    :rtype: int
    : Dynamic Programming
    """
    # 1 - 26
    # if nu[n - 1, n] \in (10, 26): F[n] = F[n - 1] + F[n - 2]
    # else: F[n] = F[n - 1]
    n = len(s)
    if n <= 1: return n
    two = int(s[:2])
    f1 = 1
    f2 = 2 if two >= 10 and two <= 26 else 1
    for i in range(2, n):
        two = int(s[i - 1: i + 1])
        if two >= 10 and two <= 26:
            f2 = f2 + f1 # forget to update f1
    return f2

numDecodings(0, '12')
```
