# 155. Min Stack


在 stack 为空的时候，我只考虑了初次初始化的情况，没考虑后面也有可能是因为 pop 而被清空。

```
class MinStack(object):

    def __init__(self):
        """
        initialize your data structure here.
        """
        self.stack = []
        self.min = None
        

    def push(self, x):
        """
        :type x: int
        :rtype: void
        """
        # if self.min is None: # In this case, if this class obj is not pushed and poped clean?
        if len(self.stack) == 0:
            self.stack.append(0)
            self.min = x
            return
        self.stack.append(x - self.min)
        if x < self.min:
            self.min = x
        

    def pop(self):
        """
        :rtype: void
        """
        if len(self.stack) == 0: return
        x = self.stack.pop()
        if x < 0:
            self.min -= x
        return 

    def top(self):
        """
        :rtype: int
        """
        if len(self.stack) == 0: return
        
        x = self.stack[-1]
        if x < 0:
            return self.min
        else:
            return x + self.min
        

    def getMin(self):
        """
        :rtype: int
        """
        return self.min
```
