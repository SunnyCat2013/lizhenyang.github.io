```
class Solution(object):
    def isPalindrome(self, s):
        """
        :type s: str
        :rtype: boolord
        """

        s_filtered = [c for c in s if c.isalnum()]
        s_filtered = ''.join(s_filtered).lower()
        n = len(s_filtered)
        i = 0
        j = n - 1
        while i < j:
            if s_filtered[i] != s_filtered[j]:
                return False
            i += 1
            j -= 1

        return True
```
