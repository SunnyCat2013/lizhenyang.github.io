# 984 String Without aaa or bbb

```
class Solution(object):
    def strWithout3a3b(self, A, B):
        """
        :type A: int
        :type B: int
        :rtype: str
        """
        la = 'a'
        lb = 'b'
        if A < B:
            A, la, B, lb = B, lb, A, la
        
        na = (A + 1) / 2
        
        if na > B:
            return (la + la + lb) * B + la * (A - 2 * B)
        
        listb = [lb] * na
        i = 0
        while B - i > na:
            listb[i] += lb
            i += 1
        
        lista = [la + la] * na
        if A % 2 == 1:
            lista[-1] = la
        
        return ''.join([la + lb for la, lb in zip(lista, listb)])
        
        
```
